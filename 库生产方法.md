# 动态库：  
- gcc -fPID -o xxx.o -c xxx.c   --->xxx.o  
- gcc -share -o libxxx.so       --->libxxx.so  
- gcc -o xxx main.c -L./ -lxxx  (使用，注意：需要把动态库拷贝到linux指定的库文件存放路径)  
# 静态库：  
- gcc -o  xxx.o -c xxx.c        --->xxx.o
- ar rcs libxxx.a xxx.o         --->libxxx.a
- gcc -o xxx main.c -L./ -lxxx  (使用)
